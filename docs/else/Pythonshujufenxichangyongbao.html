# Python数据分析常用包

  • pandas库：数据读取、数据清洗、数据统计分析、数据透视表等

    import pandas as pd

# 读取Excel文件
df = pd.read_excel("文件路径")

# 数据去重
df.drop_duplicates(inplace=True)

# 数据清洗
df.dropna(inplace=True)

# 数据统计分析
df.describe()

# 数据透视表
pd.pivot_table(df, index=["列名1"], columns=["列名2"], values=["数值列"], aggfunc="sum")

  • numpy库：多维数组支持、数学运算等

    import numpy as np

# 数组去重
np.unique(array)

# 数组统计分析
np.mean(array)
np.median(array)
np.var(array)
np.std(array)

# 数组画图
import matplotlib.pyplot as plt
plt.hist(array, bins=10)
plt.boxplot(array)

  • seaborn库：数据可视化、图表美化等

    import seaborn as sns

# 箱线图
sns.boxplot(data=df, x="列名1", y="数值列")

# 热力图
sns.heatmap(data=df, cmap="RdYlBu_r")

# 散点图
sns.scatterplot(data=df, x="列名1", y="数值列")

  • matplotlib库：基础绘图、图表润色等

import matplotlib.pyplot as plt

# 折线图
plt.plot(x, y)

# 散点图
plt.scatter(x, y)

# 柱状图
plt.bar(x, y)

# 箱线图
plt.boxplot(data, labels=labels)

    from datetime import datetime

# 字符串转日期时间
datetime.strptime("2021-09-01", "%Y-%m-%d")

# 日期时间转字符串
datetime.strftime(datetime.now(), "%Y-%m-%d %H:%M:%S")
